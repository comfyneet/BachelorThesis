logic-rules      = logic-rule { logic-rule }
logic-rule       = expression '->' conclusions [ '{' certainty-factor '}' ] ';'
expression       = term [ '|' term ]
term             = factor [ '&' factor ]
factor           = fact | '(' expr ')'
conclusions      = fact [ '&' fact ]
fact             = individual-fact
individual-fact  = class-name '=' individual-name
certainty-factor = NUMBER
class-name       = IDENT
individual-name  = IDENT

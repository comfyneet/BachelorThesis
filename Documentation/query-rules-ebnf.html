EBNF definition of Query Rules

query-rules = query-rule { NEWLINE+ query-rule }

query-rule = query [ { weight } ] NEWLINE matches

query = + term { term}

term = * | , | alternative-term | optional-term | discard-term | text

alternative-term = ( text { | text } )

optional-term = [ text { | text } ]

discard-term = [ * ]

matches = match { NEWLINE match }

match = star-position : ontology-type { , ontology-type }

ontology-type = class | relation | attribute | individual

weight = NUMBER

star-position = NUMBER

text = STRING
